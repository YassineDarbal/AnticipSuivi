Private Function GetD2D3SourceSheet() As Worksheet
    On Error Resume Next
    Set GetD2D3SourceSheet = ThisWorkbook.Worksheets("D2D3_source")
    On Error GoTo 0
    If GetD2D3SourceSheet Is Nothing Then
        Err.Raise vbObjectError + 510, "GetD2D3SourceSheet", "La feuille interne D2D3_source est introuvable. Créez-la avec la requête Power Query D2D3."
    End If
End Function

Private Sub RefreshD2D3Source(ByVal ws As Worksheet)
    Dim lo As ListObject
    Dim qt As QueryTable
    Dim refreshed As Boolean

    On Error GoTo RefreshFail
    For Each lo In ws.ListObjects
        Set qt = Nothing
        On Error Resume Next
        Set qt = lo.QueryTable
        On Error GoTo RefreshFail
        If Not qt Is Nothing Then
            qt.Refresh BackgroundQuery:=False
            refreshed = True
        End If
    Next lo

    For Each qt In ws.QueryTables
        qt.Refresh BackgroundQuery:=False
        refreshed = True
    Next qt

    If Not refreshed Then
        ThisWorkbook.RefreshAll
        Application.CalculateUntilAsyncQueriesDone
    End If
    Exit Sub

RefreshFail:
    Err.Raise vbObjectError + 511, "RefreshD2D3Source", "Impossible d'actualiser D2D3_source : " & Err.Description
End Sub

Private Function CleanD2D3Reference(ByVal RefValue As String) As String
    RefValue = KeepTextAfterToken(RefValue, "DQN ")
    RefValue = KeepTextAfterToken(RefValue, "DQN")
    RefValue = KeepTextAfterToken(RefValue, "ANTICIPATION ")
    CleanD2D3Reference = RefValue
End Function

Private Function KeepTextAfterToken(ByVal Value As String, ByVal Token As String) As String
    Dim Pos As Long

    Pos = InStr(1, Value, Token, vbTextCompare)
    If Pos > 0 Then
        KeepTextAfterToken = Mid(Value, Pos + Len(Token))
    Else
        KeepTextAfterToken = Value
    End If
End Function

Private Function RequiredHeaderColumn(ByVal headerRange As Range, ByVal headerName As String) As Long
    Dim foundHeader As Range
    Set foundHeader = headerRange.Find(headerName, LookAt:=xlWhole)
    If foundHeader Is Nothing Then
        Err.Raise vbObjectError + 512, "RequiredHeaderColumn", "Colonne obligatoire absente dans D2D3_source : " & headerName
    End If
    RequiredHeaderColumn = foundHeader.Column
End Function

Private Sub AddDictionaryValues(ByVal dict As Object, ByVal sourceRange As Range)
    Dim constants As Range
    Dim c As Range

    If Application.WorksheetFunction.CountA(sourceRange) = 0 Then Exit Sub

    On Error Resume Next
    Set constants = sourceRange.SpecialCells(xlCellTypeConstants, 23)
    On Error GoTo 0
    If constants Is Nothing Then Exit Sub

    For Each c In constants
        If Len(c.Value) > 0 Then
            If Not dict.Exists(c.Value) Then dict.Add c.Value, c.Address
        End If
    Next c
End Sub
Function DqnStatus(T, ByVal Crit1$, Optional C1% = 1, Optional Res3% = 4, Optional Res2% = 4, Optional Res1% = 5)

Dim i&, J&, K&, Temp(0 To 3), Boule As Boolean
  For i = LBound(T) To UBound(T)
    If T(i, C1) = Crit1 Then
        Temp(0) = T(i, Res1): Temp(1) = T(i, Res2): Temp(2) = T(i, Res3): Temp(3) = " "
        Exit For
    End If
  Next i
DqnStatus = Temp
End Function

Function CheckDel(W, ByVal Crit1$, ByVal Crit2$, Optional C1% = 1, Optional C2% = 1, Optional Res1% = 5)

Dim i&, J&, K&, Temps(0 To 1), Boule As Boolean
  For i = LBound(W) To UBound(W)
    If W(i, C1) = Crit1 Then
     If W(i, C2) = Crit2 Then
        Temps(0) = W(i, Res1): Temps(1) = " "
        Exit For
     End If
    End If
  Next i
CheckDel = Temps
End Function
